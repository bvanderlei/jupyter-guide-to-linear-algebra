print("B: \n", B, '\n')

# U is the upper triangular matrix
U = lag.RowAdd(B, 0, 2, 2)
F2 = lag.RowAdd(I, 0, 2, 2)

print("U: \n", U, '\n')

L1 = sla.inv(F1)
L2 = sla.inv(F2)

L = L1@L2

print("L: \n", L, '\n')

print("LU: \n", L@U, '\n')
